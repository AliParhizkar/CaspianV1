  < GridColumn AggregateField ='t => t.Key.CustomerId == null ? "---" :

t.Key.Customer .CompanyName ' Title ="Customer" />

Title ="Maximum amount" />

  < GridColumn AggregateField ="t => t.Min(u => u.TotalAmount)"

    </Columns>
