Download:

@inherits BasePage

@inject IHostEnvironment host

 

<div class="col-md-6 text-center">

<button class="c-btn c-outline-success" @onclick="async () => await DownloadFile()">Download</button>

</div>

 

@code

{

       async Task DownloadFile()

       {

           var path = host.ContentRootPath + "\\Data\\resumes.pdf";

           var byteArray = await File.ReadAllBytesAsync(path);

           await base.DownloadFile("resumes.pdf", byteArray);

    }

}
